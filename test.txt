function Write-TestHeader {
    param([string]$Title)
    Write-Host "`n========================================" -ForegroundColor $Info
    Write-Host "  $Title" -ForegroundColor $Info
    Write-Host "========================================`n" -ForegroundColor $Info
}

function Write-TestResult {
    param(
        [string]$TestName,
        [bool]$Passed,
        [string]$Message = ""
    )
    $symbol = if ($Passed) { "✅" } else { "❌" }
    $color = if ($Passed) { $Success } else { $Error }
    Write-Host "  $symbol $TestName" -ForegroundColor $color
    if ($Message) { Write-Host "     $Message" -ForegroundColor Gray }
}

function Invoke-TestRequest {
    param(
        [string]$Method,
        [string]$Path,
        [object]$Body = $null,
        [hashtable]$Headers = @{},
        [string]$CookieFile = $null
    )
    
    $url = "$API_BASE/api$Path"
    $args = @("-s", "-X", $Method, $url)
    
    if ($Headers.Count -gt 0) {
        foreach ($kvp in $Headers.GetEnumerator()) {
            $args += "-H", "$($kvp.Key): $($kvp.Value)"
        }
    }
    
    if ($Body) {
        if ($Body -is [string]) {
            $args += "-d", $Body
        } else {
            $json = $Body | ConvertTo-Json -Depth 10 -Compress
            $args += "-H", "Content-Type: application/json", "-d", $json
        }
    }
    
    if ($CookieFile) {
        $args += "-b", $CookieFile, "-c", $CookieFile
    }
    
    try {
        $response = curl.exe @args 2>$null
        $statusCode = $LASTEXITCODE
        
        # Parse response
        if ($response -and $response.StartsWith("{")) {
            return $response | ConvertFrom-Json -ErrorAction SilentlyContinue
        }
        return @{ raw = $response; statusCode = $statusCode }
    } catch {
        return @{ error = $_.Exception.Message }
    }
}
